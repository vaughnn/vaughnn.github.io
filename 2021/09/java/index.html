启动Java应用程序shell

Written by vaughn on 2021-09-01 17:47:48 in Java
| 共145字，阅读约 1 分钟 
#!/bin/bash
project=*.jar
echo "拷贝文件"
rm -rf /data/jar/$project
cp  /root/jenkins/workspace/*/target/$project /data/jar
echo "切换路径"
cd /data/jar
 
pid=`ps aux | grep java| grep $project | awk '{print $2}'`
if [ -n "$pid" ]
then
   kill -9 $pid
   echo "杀死存在进程"
fi
BUILD_ID=dontKillMe 
echo "执行"
#nohup java -jar $project > /dev/null  2>&1 &
nohup java -Xms128m -Xmx256m -jar $project > api.log  2>&1 &
echo "启动成功！"
sleep 5

执行命令

clean  package  -Dmaven.test.skip=true
clean  package -Pprod -U -Dmaven.test.skip=true
sh -x /data/startup.sh *.jar 8080
